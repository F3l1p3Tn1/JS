window.document.write('1- Hello, World!<br>')

window.document.write('3- window.navigator.cookieEnabled : ' + window.navigator.cookieEnabled + '<br>')
window.document.write('3- window.navigator.doNotTrack : ' + window.navigator.doNotTrack + '<br>')

var p0 = window.document.getElementsByTagName('p')[0]
var p1 = window.document.getElementsByTagName('p')[1]
var p2 = window.document.getElementsByTagName('p')[2]
var p3 = window.document.getElementsByTagName('p')[3]
var p4 = window.document.getElementsByTagName('p')[4]
p0.style.color = 'blue'
p1.style.color = 'green'
p2.style.color = 'yellow'
window.document.write('5- parágrafos: <br>p0- '

+ p3.innerText + '<br>p3- '
+ p3.innerHTML + '<br> (p3.innerHTML ja vem formatado)')

// Usando CSS
var bg = window.document.body

// por ID
var d = window.document.getElementById('msg')

// por Classe
document.getElementsByClassName('classe')[1].innerHTML = 'Oi! :)'
