<script>
function calcular() {
var txtv = document.querySelector('input#txtvel')
var res = document.querySelector('div#res')
var vel = Number(txtv.value)
res.innerHTML = `<p>Velocidade atual de <strong>${vel}</strong>Km/h</p>`
if (vel > 60) {
res.innerHTML += `<p><strong>MULTA</strong> por excesso de velocidade!</p>`
}
res.innerHTML += `<p>Use o cinto de segurança!</p>`
}

function verpais() {
var varpais = window.document.querySelector('input#txtpais')
var varres = document.querySelector('div#respais')
varpais = varpais.value
varpais = varpais.toLowerCase()
if (varpais == 'brasil') {
varres.innerHTML = '<p>Então você deve ser brasileiro.</p>'
} else {
varres.innerHTML = '<p>Então você deve ser estrangeiro.</p>'
}
}
</script>
