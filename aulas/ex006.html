/* Deixar esse trecho do código fora da função

var a = document.getElementById('area')
var b = document.getElementById('area2')

/* Substituindo por addEventListener:

<div id="area" onclick="clicar()" 
onmouseenter="entrar()" onmouseout="sair()"> */
a.addEventListener('click', clicar)
a.addEventListener('mouseenter', entrar)
a.addEventListener('mouseout', sair)

b.addEventListener('mousemove', moveu)
b.addEventListener('mousedown', baixo)
b.addEventListener('mouseup', cima)

function clicar() {
a.innerText = 'Clicou!'
a.style.background = 'blue'
a.style.boxShadow = 'inset 2px 2px 5px black'
}
function entrar() {
a.innerText = 'Entrou!'
a.style.background = 'red'
a.style.boxShadow = '2px 2px 5px black'
} 
function sair() {
a.innerText = 'Saiu!'
a.style.background = 'green'
a.style.boxShadow = '1px 1px 5px rgba(0, 0, 0, 0.445)'
}
function moveu() {
b.innerText = 'Moveu!'
b.style.background = 'gray'
b.style.boxShadow = '1px 1px 5px rgba(0, 0, 0, 0.445)'
}
function baixo() {
b.innerText = 'Apertou!'
b.style.background = 'orange'
b.style.boxShadow = 'inset 1px 1px 5px black'
} 
function cima() {
b.innerText = 'Largou!'
b.style.background = 'purple'
b.style.boxShadow = '1px 1px 5px black'
}
